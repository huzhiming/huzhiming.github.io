1. 根据字符串`aakkknn`,需得出结果`2a3k2n`

```js
var str = 'aakkknn'
var current = null
var count = 1
var result = {}
for(const x of str) {
    if(!current) {
        current = x
    }
    if(current==x) {
        count ++
    } else {
        current = null
        count = 1
    }
    result[x] = count + x // {"a":"2a","k":"3k","n":"2n"}
}
// Object.values({"a":"2a","k":"3k","n":"2n"}).join('')
var res = Object.values(result).join('') 

console.log('输出结果为：', res)

```
